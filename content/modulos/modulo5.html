2 Introducción a la consola de Python en QGIS

La consola de Python en QGIS permite ejecutar código directamente sobre el proyecto abierto:

Complementos → Consola de Python

4 Scripts para automatización de rutas y reportes

4.1 Script: generar múltiples buffers automáticamente

from qgis.core import (
    QgsVectorLayer, QgsFeature, QgsGeometry,
    QgsProject, QgsWkbTypes, QgsField
)
from PyQt5.QtCore import QVariant

def generar_buffers(nombre_capa, distancias):
    """
    Genera capas de buffer para una capa de puntos
    con múltiples distancias.
    """
    capa_origen = QgsProject.instance().mapLayersByName(nombre_capa)
    if not capa_origen:
        print(f"No se encontró la capa: {nombre_capa}")
        return

    capa_origen = capa_origen[0]

    for distancia in distancias:
        # Crear nueva capa de polígonos
        crs = capa_origen.crs().authid()
        capa_buffer = QgsVectorLayer(
            f"Polygon?crs={crs}",
            f"Buffer_{distancia}m",
            "memory"
        )
        provider = capa_buffer.dataProvider()
        provider.addAttributes([QgsField("nombre", QVariant.String)])
        capa_buffer.updateFields()

        entidades = []
        for entidad in capa_origen.getFeatures():
            buffer_geom = entidad.geometry().buffer(distancia, 8)
            nueva_entidad = QgsFeature()
            nueva_entidad.setGeometry(buffer_geom)
            nueva_entidad.setAttributes([entidad["nombre"]])
            entidades.append(nueva_entidad)

        provider.addFeatures(entidades)
        QgsProject.instance().addMapLayer(capa_buffer)
        print(f"Buffer de {distancia}m creado: {capa_buffer.featureCount()} entidades")

# Ejecutar la función
generar_buffers("Hospitales", [500, 1000, 2000])

4.2 Script: exportar atributos a Excel

import csv
from pathlib import Path

def exportar_a_csv(nombre_capa, ruta_salida):
    """Exporta los atributos de una capa a CSV."""
    capa = QgsProject.instance().mapLayersByName(nombre_capa)
    if not capa:
        print(f"Capa no encontrada: {nombre_capa}")
        return

    capa = capa[0]
    campos = [campo.name() for campo in capa.fields()]

    with open(ruta_salida, "w", newline="", encoding="utf-8") as f:
        writer = csv.writer(f)
        writer.writerow(campos)  # Encabezados
        for entidad in capa.getFeatures():
            writer.writerow(entidad.attributes())

    print(f"Exportado exitosamente: {ruta_salida}")

# Ejecutar
exportar_a_csv("Municipios", "/tmp/municipios.csv")

4.3 Script: exportar mapa como imagen

from qgis.core import QgsLayoutExporter, QgsProject
from qgis.utils import iface

def exportar_mapa_png(nombre_layout, ruta_salida):
    """Exporta un layout de mapa a PNG."""
    manager = QgsProject.instance().layoutManager()
    layout = manager.layoutByName(nombre_layout)

    if not layout:
        print(f"Layout no encontrado: {nombre_layout}")
        return

    exporter = QgsLayoutExporter(layout)
    settings = QgsLayoutExporter.ImageExportSettings()
    settings.dpi = 300

    resultado = exporter.exportToImage(ruta_salida, settings)
    if resultado == QgsLayoutExporter.Success:
        print(f"Mapa exportado: {ruta_salida}")
    else:
        print("Error al exportar el mapa")

# Ejecutar
exportar_mapa_png("Mi Layout", "/tmp/mapa_final.png")

5 Exportación de resultados

5.1 Exportar capa a diferentes formatos

from qgis.core import QgsVectorFileWriter, QgsCoordinateTransformContext

def exportar_capa(nombre_capa, ruta_salida, formato="ESRI Shapefile"):
    """
    Exporta una capa a shapefile, GeoJSON, GeoPackage, etc.
    Formatos: "ESRI Shapefile", "GeoJSON", "GPKG", "KML"
    """
    capa = QgsProject.instance().mapLayersByName(nombre_capa)
    if not capa:
        print(f"Capa no encontrada: {nombre_capa}")
        return
    capa = capa[0]

    opciones = QgsVectorFileWriter.SaveVectorOptions()
    opciones.driverName = formato
    opciones.fileEncoding = "UTF-8"

    error, msg, _, _ = QgsVectorFileWriter.writeAsVectorFormatV3(
        capa,
        ruta_salida,
        QgsCoordinateTransformContext(),
        opciones
    )

    if error == QgsVectorFileWriter.NoError:
        print(f"Capa exportada exitosamente: {ruta_salida}")
    else:
        print(f"Error al exportar: {msg}")

# Exportar a GeoJSON
exportar_capa("Municipios", "/tmp/municipios.geojson", "GeoJSON")
# Exportar a GeoPackage
exportar_capa("Municipios", "/tmp/municipios.gpkg", "GPKG")

6.3 Crear un script de processing personalizado

Los scripts de processing pueden guardarse en:

Configuración → Opciones → Procesamiento → Scripts

Estructura básica de un script de processing:

from qgis.core import QgsProcessingAlgorithm, QgsProcessingParameterVectorLayer
from qgis.core import QgsProcessingParameterFileDestination

class MiAlgoritmo(QgsProcessingAlgorithm):

    INPUT = "INPUT"
    OUTPUT = "OUTPUT"

    def initAlgorithm(self, config=None):
        self.addParameter(QgsProcessingParameterVectorLayer(
            self.INPUT, "Capa de entrada"))
        self.addParameter(QgsProcessingParameterFileDestination(
            self.OUTPUT, "Archivo de salida", "CSV Files (*.csv)"))

    def processAlgorithm(self, parameters, context, feedback):
        capa = self.parameterAsVectorLayer(parameters, self.INPUT, context)
        salida = self.parameterAsFileOutput(parameters, self.OUTPUT, context)

        # Tu lógica aquí
        feedback.pushInfo(f"Procesando: {capa.name()}")

        return {self.OUTPUT: salida}

    def name(self):
        return "mi_algoritmo"

    def displayName(self):
        return "Mi Algoritmo Personalizado"

    def group(self):
        return "Scripts personalizados"

    def groupId(self):
        return "scripts_personalizados"

    def createInstance(self):
        return MiAlgoritmo()

```python
QgsProject.instance().addMapLayer(resultado["OUTPUT"])
print("Buffer completado.")
```

### Crear un script de processing personalizado

Los scripts de processing pueden guardarse en:
```
Configuración → Opciones → Procesamiento → Scripts
```

Estructura básica de un script de processing:

```python
from qgis.core import QgsProcessingAlgorithm, QgsProcessingParameterVectorLayer
from qgis.core import QgsProcessingParameterFileDestination

class MiAlgoritmo(QgsProcessingAlgorithm):

    INPUT = "INPUT"
    OUTPUT = "OUTPUT"

    def initAlgorithm(self, config=None):
        self.addParameter(QgsProcessingParameterVectorLayer(
            self.INPUT, "Capa de entrada"))
        self.addParameter(QgsProcessingParameterFileDestination(
            self.OUTPUT, "Archivo de salida", "CSV Files (*.csv)"))

    def processAlgorithm(self, parameters, context, feedback):
        capa = self.parameterAsVectorLayer(parameters, self.INPUT, context)
        salida = self.parameterAsFileOutput(parameters, self.OUTPUT, context)

        # Tu lógica aquí
        feedback.pushInfo(f"Procesando: {capa.name()}")

        return {self.OUTPUT: salida}

    def name(self):
        return "mi_algoritmo"

    def displayName(self):
        return "Mi Algoritmo Personalizado"

    def group(self):
        return "Scripts personalizados"

    def groupId(self):
        return "scripts_personalizados"

    def createInstance(self):
        return MiAlgoritmo()
```
